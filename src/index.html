The universal selector or * inherits box-sizing from the <html> element. This allows third party plugins to set their own box-sizing policy.

0

Wrapping text around a figure (or any other element) without using float: left; or float: right; is not possible.

Since float does not support RTL out of the box and float: inline-start; and float: inline-end; have no browser support yet, Renormalize uses JavaScript to change all float: left; rules on elements to float: right; and vice versa based on the direction set in the <html> direction attribute, eg. <dir="rtl">.

let direction = document.querySelector("html").getAttribute("dir"),

const floats = (nodeTypes, direction) => {
    const elements = document.querySelectorAll(nodeTypes);
    elements.forEach((element) => {
        if (direction == "rtl") {
            if (getComputedStyle(element).float == "left") {
                element.style.float = "right";
            } else if (getComputedStyle(element).float == "right") {
                element.style.float = "left";
            }
        } else {
            element.style.float = getComputedStyle(element).float;
        }
    });
}

                list-style-type: none;
